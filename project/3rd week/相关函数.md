```
#调用API打开其他大模型
from openai import OpenAI
client = OpenAI(
    api_key="sk-olcrkbxpwgseittqcnsqhokycublteikefzvehwhnjyslzma",
    base_url="https://api.siliconflow.cn/v1"
)
try:
    response=client.chat.completions.create(
        model="deepseek-ai/DeepSeek-V3",    
        messages=[
            {"role": "system", "content": "你是我的人工智能助手，协助我解答问题和提供信息。"},
            {"role": "user", "content": input("请输入你的问题：")}
        ],
        max_tokens=500,
        temperature=0.7
    )
    #关键字参数，Python 会根据参数名来识别，而不是根据位置。
    print(response.choices[0].message.content)

    print("\n===== 消耗信息 =====")
    print(f"消耗总 tokens: {response.usage.total_tokens}")
    print(f"输入 tokens: {response.usage.prompt_tokens}")
    print(f"输出 tokens: {response.usage.completion_tokens}")

except Exception as e:
    print(f"调用失败：{str(e)}")
    print("\n常见问题排查：")
    print("1. API Key 是否正确（检查是否有空格/换行）")
    print("2. 网络是否能访问 https://api.siliconflow.cn")
    print("3. 账号是否有可用代金券/余额（登录硅基流动控制台查看）")
```
### 1.client=OpenAI(api_key=..., base_url=...)
初始化一个 OpenAI 客户端对象，用于与 API 服务通信。
参数--[base_url]:API 服务器地址

### 2.response=client.chat.completions.create() - 调用 API 获取响应
返回值结构：
```
response.choices[0].message.content  # 模型的文字回复
response.usage.total_tokens          # 总共消耗的 tokens
response.usage.prompt_tokens         # 输入 tokens
response.usage.completion_tokens     # 输出 tokens
```
